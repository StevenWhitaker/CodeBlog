% For \forcsvlist, used in \inplacebib below
\usepackage{etoolbox}

% Add padding to fbox that encloses \bibentry calls
\setlength{\fboxsep}{1em}

% Helper function for \inplacebib
\newcommand{\inplacebibentry}[1]{\cite{#1} \bibentry{#1}. \par}

% \inplacebib accepts a comma-separated list of citation keys
\newcommand{\inplacebib}[1]{
    % Add some vertical padding above
    \vspace{1ex}
    % Center the box
    \begin{center}
        % Add some horizontal padding with 0.9\linewidth
        \fbox{\parbox{0.9\linewidth}{
            % Add some vertical spacing between each reference
            \setlength\parskip{1em}
            % Display each reference in the comma-separated input
            \forcsvlist\inplacebibentry{#1}
        }}
    \end{center}
}
